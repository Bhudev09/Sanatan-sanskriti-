const festivalDate = new Date("March 1, 2025 00:00:00").getTime();

const countdownFunction2 = setInterval(function () {
  const now = new Date().getTime();
  const distance = festivalDate - now;

  // ‡§¶‡§ø‡§®, ‡§ò‡§Ç‡§ü‡•á, ‡§Æ‡§ø‡§®‡§ü, ‡§∏‡•á‡§ï‡§Ç‡§° ‡§®‡§ø‡§ï‡§æ‡§≤‡§®‡§æ
  const days = Math.floor(distance / (1000 * 60 * 60 * 24));
  const hours = Math.floor(
    (distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)
  );
  const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
  const seconds = Math.floor((distance % (1000 * 60)) / 1000);

    days + " ‡§¶‡§ø‡§® " + hours + " ‡§ò‡§Ç‡§ü‡•á " + minutes + " ‡§Æ‡§ø‡§®‡§ü " + seconds + " ‡§∏‡•á‡§ï‡§Ç‡§° ";

  // ‡§ü‡§æ‡§á‡§Æ ‡§ñ‡§§‡•ç‡§Æ ‡§π‡•ã‡§®‡•á ‡§ï‡•á ‡§¨‡§æ‡§¶
  if (distance < 0) {
